  \begin{itemize}

        \item Non-whitespace tokens are nodes
        \item Whitespace tokens are edges
      \end{itemize}
  \end{itemize}
  \begin{center}
    \includegraphics[width=0.8\textwidth]{img/dijkstra.png}
  \end{center}
\end{frame}
